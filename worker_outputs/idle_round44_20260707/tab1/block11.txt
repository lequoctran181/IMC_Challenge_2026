bash -lc python3 - <<'PY'
import collections
rows=[]
for line in open('/mnt/data/imc/submission_index.csv'):
    if not line.strip(): continue
    n,score,acc,file=line.strip().split(',',3)
    if score=='CE': continue
    rows.append((int(n),float(score),int(acc),file))
rows2=sorted(rows,key=lambda x:x[1],reverse=True)
print('top 50')
for r in rows2[:50]: print(r)
print('count',len(rows),'counts acc',collections.Counter(r[2] for r in rows))
print('max by n')
for r in rows[-10:]: print(r)
PY
