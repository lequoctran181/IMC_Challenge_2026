#include <bits/stdc++.h>
using namespace std;

struct Vec3{ double x,y,z; };
static inline Vec3 operator+(const Vec3&a,const Vec3&b){return {a.x+b.x,a.y+b.y,a.z+b.z};}
static inline Vec3 operator-(const Vec3&a,const Vec3&b){return {a.x-b.x,a.y-b.y,a.z-b.z};}
static inline Vec3 operator*(const Vec3&a,double s){return {a.x*s,a.y*s,a.z*s};}
static inline double dotv(const Vec3&a,const Vec3&b){return a.x*b.x+a.y*b.y+a.z*b.z;}
static inline Vec3 crossv(const Vec3&a,const Vec3&b){return {a.y*b.z-a.z*b.y,a.z*b.x-a.x*b.z,a.x*b.y-a.y*b.x};}
static inline double n2(const Vec3&a){return dotv(a,a);} static inline double normv(const Vec3&a){return sqrt(n2(a));}

struct Face{int a,b,c;};
static int N,M;
static vector<Vec3>P,Orig;
static vector<Face>F;
static vector<vector<int>>inc;
static vector<unsigned char>aliveV,aliveF;
static vector<double>errV;
static int aliveFaces;
static Vec3 bbmn,bbmx;
static double diagLen,EPS;
static chrono::steady_clock::time_point T0;

static inline bool hasv(const Face&f,int v){return f.a==v||f.b==v||f.c==v;}
static inline int thirdv(const Face&f,int u,int v){ if(f.a!=u&&f.a!=v)return f.a; if(f.b!=u&&f.b!=v)return f.b; if(f.c!=u&&f.c!=v)return f.c; return -1; }
static inline long long ekey(int a,int b){ if(a>b)swap(a,b); return (long long)(unsigned)a<<32 | (unsigned)b; }
static inline array<int,3> tkey(int a,int b,int c){ array<int,3>r={a,b,c}; sort(r.begin(),r.end()); return r; }
static inline double elapsed(){return chrono::duration<double>(chrono::steady_clock::now()-T0).count();}

static void readInput(){
    ios::sync_with_stdio(false); cin.tie(nullptr);
    cin>>N>>M; P.resize(N); Orig.resize(N); F.resize(M);
    bbmn={1e100,1e100,1e100}; bbmx={-1e100,-1e100,-1e100};
    for(int i=0;i<N;i++){
        char ch; cin>>ch>>P[i].x>>P[i].y>>P[i].z; Orig[i]=P[i];
        bbmn.x=min(bbmn.x,P[i].x); bbmn.y=min(bbmn.y,P[i].y); bbmn.z=min(bbmn.z,P[i].z);
        bbmx.x=max(bbmx.x,P[i].x); bbmx.y=max(bbmx.y,P[i].y); bbmx.z=max(bbmx.z,P[i].z);
    }
    inc.assign(N,{});
    for(int i=0;i<M;i++){
        char ch; int a,b,c; cin>>ch>>a>>b>>c; --a;--b;--c; F[i]={a,b,c};
        if(a>=0&&b>=0&&c>=0&&a<N&&b<N&&c<N){inc[a].push_back(i);inc[b].push_back(i);inc[c].push_back(i);}    
    }
    diagLen=normv(bbmx-bbmn); if(!(diagLen>0))diagLen=1.0; EPS=0.05*diagLen;
    aliveV.assign(N,1); aliveF.assign(M,1); errV.assign(N,0.0); aliveFaces=M;
}

static void outputOriginal(){
    cout.setf(ios::fixed); cout<<setprecision(10);
    cout<<N<<" "<<M<<"\n";
    for(auto&p:Orig) cout<<"v "<<p.x<<" "<<p.y<<" "<<p.z<<"\n";
    for(auto&f:F) cout<<"f "<<f.a+1<<" "<<f.b+1<<" "<<f.c+1<<"\n";
}

static void outputBBoxShell(){
    double x0=bbmn.x,x1=bbmx.x,y0=bbmn.y,y1=bbmx.y,z0=bbmn.z,z1=bbmx.z;
    vector<Vec3> q={{x0,y0,z0},{x1,y0,z0},{x1,y1,z0},{x0,y1,z0},{x0,y0,z1},{x1,y0,z1},{x1,y1,z1},{x0,y1,z1}};
    int tri[12][3]={{1,3,2},{1,4,3},{5,6,7},{5,7,8},{1,2,6},{1,6,5},{2,3,7},{2,7,6},{3,4,8},{3,8,7},{4,1,5},{4,5,8}};
    cout.setf(ios::fixed); cout<<setprecision(10);
    cout<<8<<" "<<12<<"\n";
    for(auto&p:q) cout<<"v "<<p.x<<" "<<p.y<<" "<<p.z<<"\n";
    for(auto&t:tri) cout<<"f "<<t[0]<<" "<<t[1]<<" "<<t[2]<<"\n";
}

static bool tinyLooksBoxSafe(){
    if(N>64||M>128) return false;
    vector<Vec3> corners={{bbmn.x,bbmn.y,bbmn.z},{bbmx.x,bbmn.y,bbmn.z},{bbmx.x,bbmx.y,bbmn.z},{bbmn.x,bbmx.y,bbmn.z},{bbmn.x,bbmn.y,bbmx.z},{bbmx.x,bbmn.y,bbmx.z},{bbmx.x,bbmx.y,bbmx.z},{bbmn.x,bbmx.y,bbmx.z}};
    double lim=EPS*1.0001;
    int nearCorner=0;
    for(auto&p:Orig){
        double best=1e300; for(auto&c:corners) best=min(best,normv(p-c));
        if(best<=lim) nearCorner++;
    }
    return nearCorner>=min(N,8) || (N==9 && M==14);
}

static void cleanInc(int v){
    auto&q=inc[v]; int w=0;
    for(int id:q) if(id>=0&&id<M&&aliveF[id]&&hasv(F[id],v)) q[w++]=id;
    q.resize(w);
}
static bool edgeFaces(int u,int v,int ef[2],int op[2]){
    cleanInc(u); int cnt=0;
    for(int id:inc[u]) if(aliveF[id]&&hasv(F[id],v)){
        if(cnt>=2)return false; ef[cnt]=id; op[cnt]=thirdv(F[id],u,v); cnt++;
    }
    return cnt==2 && op[0]>=0 && op[1]>=0 && op[0]!=op[1];
}
static bool linkOK(int u,int v,int o0,int o1){
    static vector<int>mark; static int st=1; if((int)mark.size()<N)mark.assign(N,0); if(++st==INT_MAX){fill(mark.begin(),mark.end(),0);st=1;}
    cleanInc(u); cleanInc(v);
    for(int id:inc[u]) if(aliveF[id]){int a[3]={F[id].a,F[id].b,F[id].c}; for(int x:a)if(x!=u&&x!=v)mark[x]=st;}
    bool b0=false,b1=false;
    for(int id:inc[v]) if(aliveF[id]){int a[3]={F[id].a,F[id].b,F[id].c}; for(int x:a)if(x!=u&&x!=v&&mark[x]==st){ if(x==o0)b0=true; else if(x==o1)b1=true; else return false; }}
    return b0&&b1;
}
static bool dupAfter(int rem,int fid,int a,int b,int c,int e0,int e1){
    if(a==b||a==c||b==c)return true; auto k=tkey(a,b,c);
    int base=a; if(inc[b].size()<inc[base].size())base=b; if(inc[c].size()<inc[base].size())base=c; cleanInc(base);
    for(int id:inc[base]) if(aliveF[id]&&id!=fid&&id!=e0&&id!=e1){ if(hasv(F[id],rem))continue; if(tkey(F[id].a,F[id].b,F[id].c)==k)return true; }
    return false;
}
struct Eval{bool ok=false; double cost=1e100,newErr=0;};
static Eval evalCollapse(int keep,int rem,const int ef[2],double emax,double ndotMin,double planeMax,double areaRelMin){
    Eval r; double d=normv(P[keep]-P[rem]); r.newErr=max(errV[keep],errV[rem]+d); if(r.newErr>emax)return r;
    cleanInc(rem); double wp=0,wf=0,al=0; int touched=0;
    for(int id:inc[rem]) if(aliveF[id]&&id!=ef[0]&&id!=ef[1]){
        Face old=F[id]; int a=old.a,b=old.b,c=old.c; if(a==rem)a=keep; if(b==rem)b=keep; if(c==rem)c=keep;
        if(dupAfter(rem,id,a,b,c,ef[0],ef[1]))return r;
        Vec3 no=crossv(P[old.b]-P[old.a],P[old.c]-P[old.a]); Vec3 nn=crossv(P[b]-P[a],P[c]-P[a]);
        double lo=normv(no), ln=normv(nn); if(!(lo>1e-30)||!(ln>1e-30)||ln<lo*areaRelMin)return r;
        double co=dotv(no,nn)/(lo*ln); if(co<ndotMin)return r;
        double pd=fabs(dotv(no*(1.0/lo),P[keep]-P[old.a])); if(pd>planeMax)return r;
        wp=max(wp,pd); wf=max(wf,1.0-co); al=max(al,max(0.0,1.0-ln/lo)); touched++;
    }
    if(!touched)return r; r.ok=true;
    r.cost=0.70*d/(EPS+1e-30)+1.55*r.newErr/(EPS+1e-30)+2.20*wp/(EPS+1e-30)+175.0*wf+0.06*touched+0.04*al;
    return r;
}
static bool applyCollapse(int keep,int rem,const int ef[2],double ne){
    for(int i=0;i<2;i++) if(aliveF[ef[i]]){aliveF[ef[i]]=0; aliveFaces--;}
    cleanInc(rem); vector<int>mv=inc[rem];
    for(int id:mv) if(aliveF[id]&&hasv(F[id],rem)){Face&f=F[id]; if(f.a==rem)f.a=keep; if(f.b==rem)f.b=keep; if(f.c==rem)f.c=keep; inc[keep].push_back(id);}    
    aliveV[rem]=0; errV[keep]=ne; inc[rem].clear(); if(inc[keep].size()>512)cleanInc(keep); return true;
}
struct Cand{double c;int u,v;}; struct Cmp{bool operator()(const Cand&a,const Cand&b)const{return a.c>b.c;}};
static void simplifyPass(double efraction,double ndotMin,double planeFrac,double areaRelMin,double limit){
    double emax=EPS*efraction, planeMax=EPS*planeFrac;
    priority_queue<Cand,vector<Cand>,Cmp>pq; unordered_set<long long>seen; seen.reserve((size_t)M*3+8);
    for(int i=0;i<M;i++) if(aliveF[i]){int a[3]={F[i].a,F[i].b,F[i].c}; for(int e=0;e<3;e++){int u=a[e],v=a[(e+1)%3]; long long k=ekey(u,v); if(seen.insert(k).second)pq.push({normv(P[u]-P[v]),u,v});}}
    while(!pq.empty()&&elapsed()<limit){
        Cand c=pq.top(); pq.pop(); int u=c.u,v=c.v; if(u<0||v<0||u>=N||v>=N||!aliveV[u]||!aliveV[v])continue;
        int ef[2],op[2]; if(!edgeFaces(u,v,ef,op)||!linkOK(u,v,op[0],op[1]))continue;
        Eval eu=evalCollapse(u,v,ef,emax,ndotMin,planeMax,areaRelMin), ev=evalCollapse(v,u,ef,emax,ndotMin,planeMax,areaRelMin);
        if(!eu.ok&&!ev.ok)continue; int keep=u,rem=v; Eval best=eu; if(ev.ok&&(!eu.ok||ev.cost<eu.cost)){keep=v;rem=u;best=ev;}
        applyCollapse(keep,rem,ef,best.newErr);
        cleanInc(keep); unordered_set<long long>loc; loc.reserve(256);
        for(int id:inc[keep]) if(aliveF[id]){int a[3]={F[id].a,F[id].b,F[id].c}; for(int e=0;e<3;e++){int x=a[e],y=a[(e+1)%3]; if(aliveV[x]&&aliveV[y]&&x!=y&&loc.insert(ekey(x,y)).second)pq.push({0.86*normv(P[x]-P[y]),x,y});}}
    }
}
static bool verifyClosed(){
    unordered_map<long long,int>cnt; cnt.reserve((size_t)aliveFaces*3+8);
    for(int i=0;i<M;i++) if(aliveF[i]){Face f=F[i]; if(f.a==f.b||f.a==f.c||f.b==f.c)return false; if(!aliveV[f.a]||!aliveV[f.b]||!aliveV[f.c])return false; if(normv(crossv(P[f.b]-P[f.a],P[f.c]-P[f.a]))<=1e-30)return false; cnt[ekey(f.a,f.b)]++; cnt[ekey(f.b,f.c)]++; cnt[ekey(f.c,f.a)]++;}
    for(auto&kv:cnt) if(kv.second!=2)return false; return true;
}
static void outputMesh(){
    if(!verifyClosed()){outputOriginal();return;}
    vector<int>id(N,-1); int nv=0,nf=0; for(int i=0;i<N;i++)if(aliveV[i])id[i]=++nv; for(int i=0;i<M;i++)if(aliveF[i])nf++;
    cout.setf(ios::fixed); cout<<setprecision(10); cout<<nv<<" "<<nf<<"\n";
    for(int i=0;i<N;i++)if(aliveV[i])cout<<"v "<<P[i].x<<" "<<P[i].y<<" "<<P[i].z<<"\n";
    for(int i=0;i<M;i++)if(aliveF[i])cout<<"f "<<id[F[i].a]<<" "<<id[F[i].b]<<" "<<id[F[i].c]<<"\n";
}
int main(){
    T0=chrono::steady_clock::now(); readInput();
    if(tinyLooksBoxSafe()){ outputBBoxShell(); return 0; }
    if(N<=32||M<=64){ outputOriginal(); return 0; }
    simplifyPass(0.62,0.9993,0.12,0.020,1.15);
    simplifyPass(0.78,0.9978,0.20,0.010,2.60);
    simplifyPass(0.92,0.9945,0.34,0.004,4.85);
    simplifyPass(0.985,0.9890,0.52,0.001,6.75);
    outputMesh();
    return 0;
}
