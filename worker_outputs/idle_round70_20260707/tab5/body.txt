static inline long long edgeKey(int a,int b){ if(a>b) swap(a,b); return ( (long long)(unsigned int)a<<32 ) | (unsigned int)b; }

static void orientOut(vector<Vec3>&V, Face &f, const Vec3& ref){
    Vec3 n=crossv(V[f.b]-V[f.a], V[f.c]-V[f.a]);
    Vec3 c=(V[f.a]+V[f.b]+V[f.c])/3.0;
    if(dotv(n,c-ref)<0) swap(f.b,f.c);
}
static void addFace(vector<Vec3>&V, vector<Face>&F, int a,int b,int c, const Vec3&ref){

}

struct GridHash{
    Vec3 mn; double cell; unordered_map<long long, vector<int>> h;
    static long long pack(int x,int y,int z){
        const long long B=1048576LL, O=524288LL;
        return ((long long)(x+O)*B + (y+O))*B + (z+O);
    }
    GridHash(){}
    GridHash(const vector<Vec3>&P,double c){init(P,c);} 
    void init(const vector<Vec3>&P,double c){
        cell=max(c,1e-12); mn={1e100,1e100,1e100};
        for(auto&p:P){mn.x=min(mn.x,p.x);mn.y=min(mn.y,p.y);mn.z=min(mn.z,p.z);}        
        h.reserve(P.size()*2+10);
        for(int i=0;i<(int)P.size();++i){int x,y,z; idx(P[i],x,y,z); h[pack(x,y,z)].push_back(i);}    }
    void idx(const Vec3&p,int&x,int&y,int&z) const{
        x=(int)floor((p.x-mn.x)/cell); y=(int)floor((p.y-mn.y)/cell); z=(int)floor((p.z-mn.z)/cell);
    }
    bool hasNear(const vector<Vec3>&P,const Vec3&q,double r) const{
        int X,Y,Z; idx(q,X,Y,Z); double r2=r*r;
        for(int dx=-1;dx<=1;dx++) for(int dy=-1;dy<=1;dy++) for(int dz=-1;dz<=1;dz++){
            auto it=h.find(pack(X+dx,Y+dy,Z+dz)); if(it==h.end()) continue;
            for(int id:it->second) if(norm2(P[id]-q)<=r2) return true;
        }
        return false;
    }
};

static bool symmetricVertexCheck(const vector<Vec3>&V,double tol){
    if(V.empty()) return false;
    GridHash ghOut(V,tol);
    for(const Vec3&p:Pin) if(!ghOut.hasNear(V,p,tol)) return false;
    GridHash ghIn(Pin,tol);
    for(const Vec3&p:V) if(!ghIn.hasNear(Pin,p,tol)) return false;
    return true;
}

static bool closedEdgeCheck(const vector<Vec3>&V,const vector<Face>&F){
    if(V.empty()||F.empty()||V.size()>(size_t)N) return false;
    double eps=max(1e-28,1e-24*DIAG*DIAG);
    unordered_map<long long,int> cnt; cnt.reserve(F.size()*4+10);
    unordered_set<unsigned long long> dup; dup.reserve(F.size()*2+10);
    auto packTri=[&](array<int,3>a)->unsigned long long{
        return ((unsigned long long)(unsigned int)a[0]<<42)^((unsigned long long)(unsigned int)a[1]<<21)^(unsigned int)a[2];
    };
    for(const Face&f:F){
        if(f.a<0||f.b<0||f.c<0||f.a>=(int)V.size()||f.b>=(int)V.size()||f.c>=(int)V.size()) return false;
        if(f.a==f.b||f.a==f.c||f.b==f.c) return false;
        if(faceArea2(V,f)<=eps) return false;
        auto tk=triKey(f); unsigned long long pk=packTri(tk); if(dup.count(pk)) return false; dup.insert(pk);
        cnt[edgeKey(f.a,f.b)]++; cnt[edgeKey(f.b,f.c)]++; cnt[edgeKey(f.c,f.a)]++;
    }
    for(auto &kv:cnt) if(kv.second!=2) return false;
    return true;
}

static bool acceptCandidate(const vector<Vec3>&V,const vector<Face>&F){

}

static void makeIcosa(vector<Vec3>&V, vector<Face>&F){

static void subdivSphere(vector<Vec3>&V, vector<Face>&F){
    map<pair<int,int>,int> mid;
    auto getmid=[&](int a,int b)->int{
        if(a>b) swap(a,b); pair<int,int> k={a,b}; auto it=mid.find(k); if(it!=mid.end()) return it->second;
        int id=V.size(); V.push_back(normed((V[a]+V[b])*0.5)); mid[k]=id; return id;
    };
    vector<Face> NF; NF.reserve(F.size()*4);
    for(auto f:F){int ab=getmid(f.a,f.b), bc=getmid(f.b,f.c), ca=getmid(f.c,f.a); NF.push_back({f.a,ab,ca}); NF.push_back({f.b,bc,ab}); NF.push_back({f.c,ca,bc}); NF.push_back({ab,bc,ca});}
    F.swap(NF);
}

static Vec3 fromAxis(int ax, double u,double v,double w){
